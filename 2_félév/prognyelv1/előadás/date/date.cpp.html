#include <iostream>
#include <string>
#include <sstream>

#include "date.h"

using namespace std;

/* konstruktor */
date::date( string s )
{
    for ( int i = 0; i < s.size(); ++i )
    {
         if ( '.' == s[ i ] || '-' == s[ i ] )
             s[ i ] = ' ';
    }
    int y, m, d;
    stringstream parser( s );
    parser >> y >> m >> d;

    set( y, m, d );
}

/* módosítók */
void date::next()
{
    static const int days_in_months[] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
    ++day;

    /* TODO: szökõév */
    if ( day - 1 == days_in_months[month - 1] )
    {
        day = 1;
        ++month;
    }
    if ( 13 == month )
    {
        month = 1;
        ++year;
    }
}

void date::add( int n )
{
    for ( int i = 0; i < n; ++i )
    {
        next();
    }
}

/* input-output */
void date::read( std::istream& is )
{
    // nem lenne igazán jó:
    // is >> year >> month >> day;
    int y, m, d;
    if ( is >> y >> m >> d )
    {
        set( y, m, d );
    }
}

void date::print( std::ostream& os ) const
{
    // nem lenne igazán jó: os << year << month << day;
    os << "[ " << get_year() << "," << get_month()
                             << "," << get_day() << " ]";
}
