22  list *list::remove()
23  {
24    if ( prev ) prev->next = next;
25    if ( next ) next->prev = prev;
26    next = prev = 0;
27    return this;
28  }

30  void list::append( list *lp )
31  {
32    next = lp->next;
33    lp->next = this;
34    prev = lp;
35    if ( next ) next->prev = this;
36  }

38  void list::insert( list *lp )
39  {
40    prev = lp->prev;
41    lp->prev = this;
42    next = lp;
43    if ( prev ) prev->next = this;
44  }
