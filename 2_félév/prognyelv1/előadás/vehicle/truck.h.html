#ifndef TRUCK_H
#define TRUCK_H

#include <string>
#include "vehicle.h"

class truck : public vehicle
{
public:
  truck( const std::string& s, double w ) : vehicle( s ), weight( w ) { }

  /* virtual */ bool mv() { return vehicle::mv() && weight < 4.0; }
private:
  double weight;
};

#endif /* TRUCK_H */
